#include <SoftwareSerial.h>
    #include <DFPlayer_Mini_Mp3.h>
    
    int out1 = 2;
    int in1 = 3;
    int out2 = 4;
    int in2 = 5;
    int out3 = 6;
    int in3 = 7;
    int command;
    void setup()
    {     
          Serial.begin(9600);
          pinMode(out1,OUTPUT);
          pinMode(out2,OUTPUT);
          pinMode(out3,OUTPUT);
          mp3_set_serial (Serial);
          delay(1);
          mp3_set_volume (20);
          delay(1000);
          randomSeed(analogRead(A0));
          command=random(1, 4);
          mp3_stop();
      }
    
    void loop() {
      int a,b,c;
    
      a=digitalRead(in1);
      b=digitalRead(in2);
      c=digitalRead(in3);
      Serial.print(command);
     
      
      if(command==1){ 
        
        
        digitalWrite(out1,HIGH);
        digitalWrite(out2,LOW);
        digitalWrite(out3,LOW);
        if(a==HIGH){
        mp3_play(1);
        digitalWrite(out1,LOW);
        delay(3000);mp3_stop();
        command=random(1, 4);
        }
      }
      else if(command==2){
        digitalWrite(out1,LOW);
        digitalWrite(out2,HIGH);
        digitalWrite(out3,LOW);
        if(b==HIGH){
        mp3_play(2);
        digitalWrite(out2,LOW);
        delay(3000);mp3_stop();
        command=random(1, 4);}
        }
    
        else if(command==3){ 
        
        digitalWrite(out1,LOW);
        digitalWrite(out2,LOW);
        digitalWrite(out3,HIGH);
     
        if(c==HIGH){
        mp3_play(3);
        digitalWrite(out3,LOW);
        delay(3000);mp3_stop();
        command=random(1, 4);
        }
      }  
      
      
    }
