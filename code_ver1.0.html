#include <SoftwareSerial.h>
    #include <DFPlayer_Mini_Mp3.h>
    int led1=2;
    int led2=3;
    int led3=4;
    int led4=5;
    int bottom=A0;
    int bls=0;
    int ls=0;
    char val;
    SoftwareSerial BT(10,11);
    void setup() {
      Serial.begin(9600);
      
      mp3_set_serial (Serial);
      delay(1);
      mp3_set_volume (25);
      Serial.println("BT is ready!");
      BT.begin(9600);//bluetooth baud rate  
     pinMode(led1,OUTPUT);
     pinMode(led2,OUTPUT);
     pinMode(led3,OUTPUT);
     pinMode(led4,OUTPUT);
     pinMode(bottom,INPUT);
     randomSeed(analogRead(A1));
     delay(1000);
    }
    
    void loop() {
      int i;
      
         
      String data = "";
    
          if (BT.available()) {
                while (BT.available()) {
                val = BT.read();
                if(val!='\n'){data += val;}
                Serial.print(val);
                delay(5);}
                
                }
                
      if(data=="1"){
                    mp3_play(1);
                    digitalWrite(led1,HIGH);
                    digitalWrite(led2,HIGH);
                    digitalWrite(led3,HIGH);
                    delay(3000);
                    digitalWrite(led1,HIGH);
                    digitalWrite(led2,LOW);
                    digitalWrite(led3,LOW);
                    }
                    
                    else if(data=="3"){
                      mp3_play(2);
                    digitalWrite(led1,HIGH);
                    digitalWrite(led2,HIGH);
                    digitalWrite(led3,HIGH);
                    delay(3000);
                      digitalWrite(led1,LOW);
                     digitalWrite(led2,HIGH);
                     digitalWrite(led3,LOW); }
                     
                   else if(data=="5"){
                    mp3_play(3);
                    digitalWrite(led1,HIGH);
                    digitalWrite(led2,HIGH);
                    digitalWrite(led3,HIGH);
                    delay(3000);
                    digitalWrite(led1,LOW);
                    digitalWrite(led2,LOW);
                    digitalWrite(led3,HIGH); }
                    }
